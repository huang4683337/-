## 同步责任链

```js
function f1(ctx, next) {
    console.log('执行到f1')
    next()
    console.log('f1执行完毕')
}

function f2(ctx, next) {
    console.log('执行到f2')
    next()
    console.log('f2执行完毕')
}

function f3(ctx, next) {
    console.log('执行到f3')
    console.log('f3执行完毕')
}
```

```js
let next1 = function (ctx) {
    f1(ctx, next2)
}

let next2 = function (ctx) {
    f2(ctx, next3)
}

let next3 = function (ctx) {
    f3(ctx)
}

next1("ctx");

// 执行到f1
// 执行到f2
// 执行到f3
// f3执行完毕
// f2执行完毕
// f1执行完毕
```



## 一次性执行所有 - 同步

```js
function nextCompose(middle) {
    return function (ctx) {
        dispatch(0);
        function dispatch(i) {
            let fn = middle[i];
            if (!fn) {
                return;
            }

            fn(ctx, function () {
                dispatch(i + 1);
            })
        }
    }
}

const middle = [f1, f2, f3];
nextCompose(middle)('执行环境上下文');

// 执行到f1
// 执行到f2
// 执行到f3
// f3执行完毕
// f2执行完毕
// f1执行完毕
```



## 异步责任链

+ 异步操作需要使用到 Promise，所以添加 asycn await

  ```js
  async function f1(ctx, next) {
      console.log('执行到f1')
      await next()
      console.log('f1执行完毕')
  }
  
  async function f2(ctx, next) {
      console.log('执行到f2')
      await next()
      console.log('f2执行完毕')
  }
  
  function f3(ctx, next) {
      console.log('执行到f3')
      console.log('f3执行完毕')
  }
  ```

+ 给函数 f2 添加异步操作

  ```js
  async function f2(ctx, next) {
      console.log('执行到f2')
      await new Promise((resolve, reject) => {
          setTimeout(() => {
              console.log('setTimeout');
              resolve('setTimeout');
          }, 2000)
      })
      await next()
      console.log('f2执行完毕')
  }
  ```
  
**使用 `nextCompose`一次性执行所有查看结果**
  
```js
  function nextCompose(middle) {
      return function (ctx) {
          dispatch(0);
          function dispatch(i) {
              let fn = middle[i];
              if (!fn) {
                  return;
              }
  
              fn(ctx, function () {
                  dispatch(i + 1);
              })
          }
      }
  }
  
  const middle = [f1, f2, f3];
  nextCompose(middle)('执行环境上下文');
  
  // 执行到f1
  // 执行到f2
  // f1执行完毕
  // setTimeout
  // 执行到f3
  // f3执行完毕
  // f2执行完毕
  ```
  
执行结果，并不是我们所期望的答案  12 setTimeout 3 --- 321 ，而是 121 setTimeout 332。
  
分析运行结果，我们发现 f1 的后续在 f2 没有执行完就执行了，很明显它插队了。
  
那么可能是 f1 的 `await next()` 没有等待Promise承诺结果就执行了。
  
```js
  // 在 f1 中打印
  async function f1(ctx, next) {
      console.log('next()是一个执行承诺吗', next());	// undefined
      console.log('执行到f1')
      await next()
      console.log('f1执行完毕')
  }
  // 结果很明显，next() 函数没有返回对应函数的执行承诺
  // 我们需要给每个函数添加 Promise 执行承诺并且返回
  ```
  
+ `nextCompose` 函数中，给每一个函数添加 Promise 执行承诺，并将该承诺返回。

  目的是为了上一个函数的 await
  
  ```js
  // dispatch： 执行结果为一个 Promise 承诺
  // return dispatch(): 将这个 Promise 承诺返回给上一个函数，作为 await next() 时使用
  
  function nextCompose(middle) {
      return function (ctx) {
          return dispatch(0);    // 第一次执行也要返回当前函数执行承诺
          function dispatch(i) {
              let fn = middle[i];
              if (!fn) {
                  return;
              }
  
              // 给每个函数添加一个 Promise 执行承诺
              return Promise.resolve(
                  fn(ctx, function () {
                      return dispatch(i + 1); // 返回当前函数的执行承诺
                  })
            )
          }
      }
  }
  
  const middle = [f1, f2, f3];
  nextCompose(middle)('执行环境上下文');
  
  // 执行到f1
  // 执行到f2
  // setTimeout
  // 执行到f3
  // f3执行完毕
  // f2执行完毕
  // f1执行完毕
  ```
  
  



## 补充 - 函数中使用 promise 时如何保证承诺返回

promise 执行完毕会返回一个执行承诺 padding - resolved、padding - rejected

```js
// promise 执行完毕会返回一个执行承诺 padding - resolved、padding - rejected

const promise = new Promise((resolve, reject) => { resolve(1); })
console.log(promise);
```



如果要使用 await xx() , 需要在 xx 函数中将 Promise 执行承诺返回

```js
// 如果要使用 await xx() , 需要在 xx 函数中将 Promise 执行承诺返回

const a = () => { Promise.resolve(1); }
console.log(a());   // undefined

const b = () => Promise.resolve(1);
console.log(b());   // Promise { 1 }
```

```js
// 函数 a 中没有将 Promise 执行承诺返回
function a() {
    new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('setTimeout');
        }, 1000)
    })
}

async function promise() {
    console.log(await a());   // undefined
    console.log(1);
}

promise();
// 1 
// setTimeout
```

```js
// 函数 a 中将 Promise 执行承诺返回
function a() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('setTimeout');
        }, 1000)
    })
}

async function promise() {
    console.log(await a());   // setTimeout
    console.log(1);
}

promise();
// setTimeout 
// 1
```
