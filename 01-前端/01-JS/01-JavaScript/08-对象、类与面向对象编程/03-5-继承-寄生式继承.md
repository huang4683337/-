+ 通过工厂模式
+ `Object` 以传入的对象为模板创建一个新对象。
+ 给新对象进行扩展
+ 返回新对象

```js
function createAnoter(obj) {
    let clone = Object(obj)
    clone.say = function () {
        console.log('hi');
    }
    return clone;
}

let person = {
    name: 'tom',
    friends: [1, 2]
}

let anotherPerson = createAnoter(person);

anotherPerson.say();    // hi
console.log(anotherPerson.name);    // tom
```
