## 准备工作

**根目录创建 webpack.mpa.config.js **



**新建 src/mpa 文件夹**

```shell
|- mpa
	|- index
		|- index.html
		|- index.js
	|- list
		|- index.html
		|- index.js
	|- detail
		|- index.html
		|- index.js
```



**通过通配符匹配想要的文件路径**

```shell
$ npm i glob -D
```



**package.json 添加多页面打包命令**

```json
  "scripts": {
    "//mpa": "执行多页面配置的 webpack 文件",
    "mpa": "webpack --config ./webpack.mpa.config.js"
  },
```



## 多页面 mpa 生成函数

**webpack.config.js**

```js
/**
 * 使用函数生成 webpack 配置的 entry htmlwebpackplugin
 */

const glob = require("glob");

// 返回 entry、 htmlWebpackPlugin
const setMpa = () => {
    const entry = {};
    const htmlwebpackplugin = [];

    // 获取符合条件的 入口文件
    const entryFiles = glob.sync(path.join(__dirname, "./src/mpa/*/index.js"));

    entryFiles.map((item, index) => {
        const entryFile = item;

        // 获取目录名称
        const match = entryFile.match(/src\/mpa\/(.*)\/index.js$/);
        const pageName = match[1];

        //生成入口 entry
        entry[pageName] = entryFile;

        // 生成 htmlwebpackplugin
        htmlwebpackplugin.push(new HtmlWebpackPlugin({
            template: path.join(__dirname, `./src/mpa/${pageName}/index.html`),
            filename: `${pageName}.html`,
            chunks: [pageName],
        }))
    })
    return {
        entry,
        htmlwebpackplugin
    }
}

// 解构
const { entry, htmlwebpackplugin } = setMpa();
```

```js
module.exports = {
    // entry:入口
    entry,

    // output:出口
    output: {
        path: path.resolve(__dirname, "./mpa"),
        filename: "[name].js",
    },
  
  	...
  	
		plugins: [new CleanWebpackPlugin(), ...htmlwebpackplugin],
}
```



## 构建

```shell
$ npm run mpa
# 发现根目录多了一个 mpa 文件
|- mpa
	|- detail.js
	|- detail.html
	|- index.js
	|- index.html
	|- list.js
	|- list.html
```



```shell
# webpack.config.js 中
# filename: `${pageName}.html`,
# 替换成
# filename: `${pageName}/${pageName}.html`,

# filename: "[name].js",
# 替换成
# filename: "[name]/[name].js",

$ npm run mpa

|- mpa
	|- index
		|- index.html
		|- index.js
	|- list
		|- list.html
		|- list.js
	|- detail
		|- detail.html
		|- detail.js
```
