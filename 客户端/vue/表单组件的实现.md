## 实现KInput

```html
<template>
  <div>
    <!-- 自定义组件双绑：v-model语法糖，:value，@input -->
    <input :type="type" :value="value" @input="onInput" v-bind="$attrs">
  </div>
</template>

<script>
  export default {
    inheritAttrs: false,
    props: {
      value: {
        type: String,
        default: ''
      },
      type: {
        type: String,
        default: 'text'
      }
    },
    methods: {
      onInput(e) {
        this.$emit('input', e.target.value)
        
        // 触发校验
        this.$parent.$emit('validate')
      }
    },
  }
</script>
```
