# 武汉辰安项目总结

## 路由的使用

+ 新建一个当前模块的文件夹

  ```js
  + src
    + newFile
      + controller	
  	  + b
  		b.ts
  	  a.ts
  	+ style
  	+ template
  	  + b
  		b.html
  	  a.html
  	newFileModule.ts
  ```

  ```js
  // newFileModule.ts
  import { SimpleModule } from 'prism-web';
  
  
  import { aTS } from './controller/a';	
  
  class A extends SimpleModule {
      constructor(){
          super()
      }
      
      configureRouter() {
          this.addRouter('浏览器地址', process.env.baseurl+'html文件地址', resolve => require(['逻辑层地址'], resolve), true,'b')
          this.addRouter("/",'/newFile/a') // 默认页面
          this.addRouter('/a/b', process.env.baseurl+'newFile/template/b/b.html', resolve => require(['./controller/b'], resolve), true,'b')
      }
      
      // 配置弹窗
      configureView(){
          this.addView( process.env.baseurl+'newFile/template/a.html', aTS, 'a')
      }
  }
  ```

  

+ 在 main.ts 中注册

  ```js
  const NewFileModule = require('./NewFileModule/newFileModule')
  
  const modules = [NewFileModule]
  ```
